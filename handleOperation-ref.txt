// Make sure display field has a value before symbol is pressed

if (disp.value.length > 0) {
	disp.value += (emptyStr + e.target.dataset.value);
}

// Make sure current symbol doesn't repeat itself

if (disp.value.split('')[(disp.value.length - 1)] === disp.value.split('')[disp.value.length - 2]) {
	disp.value = disp.value.substr(0, disp.value.length - 1);
}

if (calculator.model.isAcceptedSym(disp.value.split('')[disp.value.length - 2])) {
	tempArr = disp.value.split('');
	tempArr.splice( (tempArr.length - 2) , 1);
	disp.value = '';
	disp.value += (emptyStr + tempArr.join(''));
}
